// 生成调色板 function generatePalette(colors, paletteName) { paletteContainer.innerHTML = ''; colors.forEach(color => { const colorBox = document.createElement('div'); colorBox.className = 'palette-color'; colorBox.style.backgroundColor = color; colorBox.title = color; colorBox.addEventListener('click', function() { colorPicker.value = color; colorDisplay.style.backgroundColor = color; updateColorInfo(color); }); paletteContainer.appendChild(colorBox); }); // 生成调色板代码 let code = `/* ${paletteName} */\n`; colors.forEach((color, index) => { code += `--color-${index + 1}: ${color};\n`; }); paletteCode.textContent = code; } // 复制功能 document.getElementById('copyHex').addEventListener('click', function() { copyToClipboard(hexValue.textContent); }); document.getElementById('copyRgb').addEventListener('click', function() { copyToClipboard(rgbValue.textContent); }); document.getElementById('copyHsl').addEventListener('click', function() { copyToClipboard(hslValue.textContent); }); function copyToClipboard(text) { navigator.clipboard.writeText(text).then(() => { alert('已复制: ' + text); }).catch(err => { console.error('复制失败: ', err); }); }
